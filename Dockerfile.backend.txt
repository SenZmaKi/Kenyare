FROM python:3.12-slim AS builder

WORKDIR /kenyare

# Install build dependencies
RUN apt-get update && apt-get install -y \
    build-essential \
    poppler-utils \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements
COPY kenyare/requirements.txt ./requirements.txt

# Install dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Production image
FROM python:3.12-slim

WORKDIR /kenyare

# Install runtime dependencies and remove apt list cache
RUN apt-get update && apt-get install -y \
    poppler-utils \
    && rm -rf /var/lib/apt/lists/*

# Copy dependencies from builder
COPY --from=builder /usr/local/lib/python3.12/site-packages/ /usr/local/lib/python3.12/site-packages/
COPY --from=builder /usr/local/bin/ /usr/local/bin/

# Copy application
COPY . .

# Create non-root user
RUN useradd -m -U kenyare_user && \
    chown -R kenyare_user:kenyare_user /kenyare

USER kenyare_user

EXPOSE 8000

HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')"

CMD ["python", "-m", "kenyare.server"]

# Add a label for the image
LABEL name="kenyare-backend" version="1.0"
