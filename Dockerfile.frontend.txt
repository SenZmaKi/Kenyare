FROM node:20-alpine

# Install system dependencies
RUN apk add --no-cache \
    poppler-utils \
    wget \
    && rm -rf /var/cache/apk/*

WORKDIR /kenyare

# Install pnpm globally before switching to non-root user
RUN npm install -g pnpm

# Create non-root user
RUN addgroup -S appgroup && adduser -S appuser -G appgroup

# Set proper ownership of the working directory and npm global directory
RUN mkdir -p /home/appuser/.npm && \
    chown -R appuser:appgroup /home/appuser/.npm && \
    chown -R appuser:appgroup /kenyare

# Switch to non-root user
USER appuser

# Copy package files with correct ownership
COPY --chown=appuser:appgroup package.json pnpm-lock.yaml* ./

# Install dependencies as non-root user
RUN pnpm install

# Copy the rest of the application with correct ownership
COPY --chown=appuser:appgroup . .

# Ensure temp directory exists and has correct permissions
RUN mkdir -p /kenyare/node_modules/.vite && \
    mkdir -p /kenyare/node_modules/.pnpm

EXPOSE 5173

# Run in dev mode with host and port explicitly set
CMD ["sh", "-c", "pnpm run dev -- --host 0.0.0.0 --port 5173"]

# Image metadata
LABEL name="kenyare-frontend" version="1.0"
