FROM node:20-alpine

# Install system dependencies
RUN apk add --no-cache \
    poppler-utils \
    wget \
    && rm -rf /var/cache/apk/*

WORKDIR /kenyare

# Create non-root user first
RUN addgroup -S kenyare_group && adduser -S kenyare_user -G kenyare_group

# Set proper ownership of the working directory
RUN chown -R kenyare_user:kenyare_group /kenyare

# Switch to non-root user
USER kenyare_user

# Copy package files with correct ownership
COPY --chown=kenyare_user:kenyare_group package.json pnpm-lock.yaml* ./

# Install pnpm and dependencies
RUN npm install -g pnpm && \
    pnpm install

# Copy the rest of the application with correct ownership
COPY --chown=kenyare_user:kenyare_group . .

# Ensure temp directory exists and has correct permissions
RUN mkdir -p /kenyare/node_modules/.vite && \
    chown -R kenyare_user:kenyare_group /kenyare/node_modules/.vite

EXPOSE 5173

# Run in dev mode with host and port explicitly set
CMD ["sh", "-c", "pnpm run dev -- --host 0.0.0.0 --port 5173"]

# Image metadata
LABEL name="kenyare-frontend" version="1.0"
